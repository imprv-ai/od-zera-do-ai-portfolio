# zdefiniujmy sobie kilka funkcji-helperw
def format_corr_value(val):
    if val == 1.0:
        return '1'
    v = int(val * 100)
    av = (abs(v) // 10) - 4
    if (av > 0):
        return '%s%s' % (v, '!' * av)
    return str(v)

def format_corr(dt):
    dt2 = dt.copy()
    for c in dt2.columns:
        dt2[c] = dt2[c].apply(format_corr_value)
    return dt2
